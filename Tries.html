Trie

Learn about Tries (Prefix Trees) and their efficient string operations

What is a Trie?

A Trie (prefix tree) is a tree-like data structure that stores strings for efficient prefix searches. It's commonly used for autocomplete systems and dictionary implementations.

Key Characteristics:

  • Efficient prefix searches (O(m) where m is search string length)
  • Space efficient for storing many strings with common prefixes
  • Each node represents a character in the string
  • End nodes mark complete words
  • Faster than hash tables for prefix-based operations

Basic Operations:

Insert

function insert(word) {
  let node = root;
  for (const char of word) {
    if (!node.children[char]) {
      node.children[char] = new TrieNode();
    }
    node = node.children[char];
  }
  node.isEnd = true;
}

Search

function search(word) {
  let node = root;
  for (const char of word) {
    if (!node.children[char]) {
      return false;
    }
    node = node.children[char];
  }
  return node.isEnd;
}

Prefix Search

function startsWith(prefix) {
  let node = root;
  for (const char of prefix) {
    if (!node.children[char]) {
      return false;
    }
    node = node.children[char];
  }
  return true;
}

Delete

function delete(word) {
  if (!search(word)) return false;
  
  let node = root;
  const stack = [];
  for (const char of word) {
    stack.push([node, char]);
    node = node.children[char];
  }
  node.isEnd = false;
  
  // Clean up unused nodes
  while (stack.length > 0) {
    const [parent, char] = stack.pop();
    const node = parent.children[char];
    if (Object.keys(node.children).length === 0 && !node.isEnd) {
      delete parent.children[char];
    } else {
      break;
    }
  }
  return true;
}
